More than one controls

Code

Html

<span data-sc-id="SimpleTextControl" data-bind="text:text" class="sc-text"></span>
<span data-sc-id="SimpleTextControl2" data-bind="text:text" class="sc-text"></span>

Javascript

_sc.Factories.createBaseComponent({
  name: "TextControl",
  selector: ".sc-text",
  attributes: [ 
    { name: "text", value: "$el.text" }
  ]
 });

window.app = _sc.Factories.createApp();
