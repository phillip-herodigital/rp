More than one controls

Code

Html

<span data-sc-id="SimpleTextControl" data-bind="text:text" class="sc-text"><A simple text conrol</span>
<button data-sc-id="Button" data-bind="text:text" class="sc-button">Do something</button>

Javascript

_sc.Factories.createBaseComponent({
  name: "TextControl",
  selector: ".sc-text",
  attributes: [ 
    { name: "text", value: "$el.text" }
  ]
 });

_sc.Factories.createBaseComponent({
  name: "Button",
  selector: ".sc-button",
  attributes: [ 
    { name: "text", value: "$el.text" }
  ],
  events: {
    "click": "doSomething"
  },
  doSomething: function() {
    alert(this.model.get("text"));
  }
});


window.app = _sc.Factories.createApp();

Result

A text control
